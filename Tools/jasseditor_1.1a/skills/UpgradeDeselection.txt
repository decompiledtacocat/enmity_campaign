function UpgradeDeSelectionConditions takes nothing returns boolean
return GetBooleanOr(GetTriggerUnit() == gg_unit_u003_0213 ,   GetTriggerUnit() == gg_unit_h00P_0211  ) 
endfunction
function UpgradeDeSelectionActions takes nothing returns nothing
//h00p ws, u003 sh
local player selector=GetTriggerPlayer()
local unit stupid=GetTriggerUnit()
local integer id=GetConvertedPlayerId(selector)
       if stupid==gg_unit_u003_0213 then
        call SetUnitOwner( gg_unit_u003_0213, Player(6), false )
       elseif stupid==gg_unit_h00P_0211 then
        call SetUnitOwner( gg_unit_h00P_0211, Player(1), false )
        endif
set stupid=null
set selector=null    
endfunction
