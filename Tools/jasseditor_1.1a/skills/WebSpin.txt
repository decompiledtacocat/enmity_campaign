function WebSpinConditions takes nothing returns boolean
return ( GetSpellAbilityId() == 'A05K' )
endfunction
function WebSpinActions takes nothing returns nothing
local unit target= GetSpellTargetUnit()
local unit caster=GetSpellAbilityUnit()
call SetHandleHandle(caster,"target",target)
set target=null
set caster=null
endfunction
//===========================================================================
function InitTrig_WebSpin takes nothing returns nothing
    set gg_trg_WebSpin = CreateTrigger(  )
    call TriggerRegisterAnyUnitEventBJ( gg_trg_WebSpin, EVENT_PLAYER_UNIT_SPELL_EFFECT )
    call TriggerAddCondition( gg_trg_WebSpin, Condition( function WebSpinConditions) )
    call TriggerAddAction( gg_trg_WebSpin, function WebSpinActions)
endfunction
