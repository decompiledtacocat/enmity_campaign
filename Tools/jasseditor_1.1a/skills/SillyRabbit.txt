function SillyRabbitConditions takes nothing returns boolean
local integer rabbit=GetUnitTypeId(GetDyingUnit())
return GetBooleanOr(GetBooleanOr(GetBooleanOr((rabbit == 'n00S' ),(rabbit == 'u00E' )) ,GetBooleanOr((rabbit == 'u011' ),(rabbit== 'n00Q' ))),GetBooleanOr((rabbit == 'h00G' ),GetBooleanOr(GetBooleanOr((rabbit == 'u00V' ),(rabbit == 'n00X' )),GetBooleanOr((rabbit == 'n011' ),(rabbit == 'u012' )))   ))
endfunction
function SillyRabbitActions takes nothing returns nothing
local unit rabbit=GetDyingUnit()
call RemoveUnit(rabbit)
set rabbit=null
endfunction
//===========================================================================
function InitTrig_SillyRabbit takes nothing returns nothing
    set gg_trg_SillyRabbit = CreateTrigger(  )
    call TriggerRegisterAnyUnitEventBJ( gg_trg_SillyRabbit, EVENT_PLAYER_UNIT_DEATH )
    call TriggerAddCondition( gg_trg_SillyRabbit, Condition( function SillyRabbitConditions) )
    call TriggerAddAction( gg_trg_SillyRabbit, function SillyRabbitActions)
endfunction
