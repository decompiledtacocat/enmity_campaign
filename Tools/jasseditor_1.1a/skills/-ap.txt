function RegisterConsumer takes unit c returns nothing
local integer x=0
local integer f=0

loop
exitwhen x>=50
if udg_Consumer[x]==c then
set x=50
set f=1
else
set x=x+1
endif
endloop

if f==0 then
set x=0
loop
exitwhen x>=50
if udg_Consumer[x]==null then
set udg_Consumer[x]=c
set udg_ConsumerDmg[x]=0
set udg_ConsumerBonus[x]=0
set x=50
endif
set x=x+1
endloop
endif
endfunction
