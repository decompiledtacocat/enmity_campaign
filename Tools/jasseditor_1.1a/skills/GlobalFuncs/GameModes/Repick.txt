function RepickConditions takes nothing returns boolean
return  ( udg_TimeMarker[1] == false )
endfunction
function RepickFilter takes nothing returns boolean
    return GetBooleanAnd(( IsUnitType(GetFilterUnit(), UNIT_TYPE_HERO) == true ),RectContainsUnit(gg_rct_HeroSelection, GetFilterUnit()) == false )
endfunction
function RepickActions takes nothing returns nothing
local unit picked
local player owner=GetTriggerPlayer()
local group temp=GetUnitsOfPlayerMatching(owner, Condition(function RepickFilter ))

loop
set picked=FirstOfGroup(temp)
exitwhen picked==null
call MultiCheck(picked,false)
    call SetPlayerStateBJ( owner, PLAYER_STATE_RESOURCE_GOLD, 425 )
    call SetPlayerStateBJ( owner, PLAYER_STATE_RESOURCE_LUMBER, 100 )
call RemoveUnit(picked)
call GroupRemoveUnit(temp,picked)
endloop
set picked=null
call DestroyGroup(temp)
set temp=null
set owner=null

endfunction
